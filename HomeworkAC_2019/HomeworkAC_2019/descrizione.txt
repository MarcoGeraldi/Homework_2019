module circuito1 (
input l, m
output x
);
assign x = NOT (m AND l)
endmodule

module circuito2 (
input r, n
output  y
);
assign y = r OR n
endmodule

module circuito_composto (
clk
input a, b, c, d
output o, k, w
);
circuito1 instance (.l=a, .m=b, .x=FF1)
circuito2 instance (.r=c, .n=d, .y=FF2)
assign k = FF1 OR FF2
endmodule
